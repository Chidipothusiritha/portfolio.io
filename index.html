# FilingsQA — RAG over SEC 10-K/10-Q filings
import psycopg2
from openai import OpenAI

client = OpenAI()

def retrieve_and_answer(query: str, conn) -> dict:
    vec = client.embeddings.create(
        input=query, model="text-embedding-3-small"
    ).data[0].embedding

    cur = conn.cursor()
    cur.execute("""
        SELECT chunk_text, source_file
        FROM filings_chunks
        ORDER BY embedding <=> %s::vector
        LIMIT 5
    """, (vec,))
    hits = cur.fetchall()

    context = "\n\n".join(r[0] for r in hits)
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[
            {"role": "system", "content": "Answer using only the context provided."},
            {"role": "user", "content": f"Context:\n{context}\n\nQ: {query}"}
        ]
    )
    return {"answer": response.choices[0].message.content,
            "sources": [r[1] for r in hits]}
